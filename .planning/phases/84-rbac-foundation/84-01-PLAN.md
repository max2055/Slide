```typescript
  async listRoles(): Promise<Array<...>>
  async updateRole(roleId: number, updates: { name?: string; description?: string }): Promise<{ success: boolean; error?: string }>
  async deleteRole(roleId: number): Promise<{ success: boolean; error?: string }>  // Reject if is_system=true

  // ----- Permissions -----
  async createPermission(code: string, name: string, resource: string, action: string, description?: string): Promise<{ success: boolean; permissionId?: number; error?: string }>
  async getPermission(permissionId: number): Promise<...>
  async getPermissionByCode(code: string): Promise<...>
  async listPermissions(resource?: string): Promise<Array<...>>
  async deletePermission(permissionId: number): Promise<{ success: boolean; error?: string }>

  // ----- Role-Permission assignment -----
  async assignPermissionToRole(roleId: number, permissionId: number): Promise<{ success: boolean; error?: string }>
  async revokePermissionFromRole(roleId: number, permissionId: number): Promise<{ success: boolean; error?: string }>
  async getRolePermissions(roleId: number): Promise<Array<{ id: number; code: string; name: string; resource: string; action: string }>>
  async getRolePermissionCodes(roleId: number): Promise<string[]>  // Returns just the codes

  // ----- User-Role assignment -----
  async assignRoleToUser(userId: number, roleId: number): Promise<{ success: boolean; error?: string }>
  async revokeRoleFromUser(userId: number, roleId: number): Promise<{ success: boolean; error?: string }>
  async getUserRoles(userId: number): Promise<Array<{ id: number; role_id: number; role_name: string }>>
  async getUserRoleIds(userId: number): Promise<number[]>

  // ----- Instance-Permission assignment -----
  async grantInstanceAccess(userId: number, instanceId: number): Promise<{ success: boolean; error?: string }>
  async revokeInstanceAccess(userId: number, instanceId: number): Promise<{ success: boolean; error?: string }>
  async getUserInstanceAccess(userId: number): Promise<number[]>  // Returns instance IDs
  async getUsersWithInstanceAccess(instanceId: number): Promise<number[]>  // Returns user IDs

  // ----- Permission lookup (used by middleware) -----
  async getUserPermissions(userId: number): Promise<Set<string>>
    // JOIN: user_roles -> role_permissions -> permissions
    // Returns a Set of permission codes (e.g., "instance:view")
    // When admin role has '*' wildcard assigned, the JOIN returns Set(['*'])

  async checkInstanceAccess(userId: number, instanceId: number): Promise<boolean>
    // SELECT COUNT(*) FROM instance_permissions WHERE user_id = ? AND instance_id = ?
    // Returns true if > 0
}
```

**Implementation details:**
- All SQL queries use parameterized `pool.execute('SELECT ... WHERE ? = ?', [val])` pattern
- All methods that query by `roles.name` use UNIQUE index lookup

- `deletePermission` checks if any role references it (SELECT from role_permissions first)
- `getUserPermissions` performs the 3-table JOIN: `user_roles -> role_permissions -> permissions`
- `checkInstanceAccess` does a simple `SELECT COUNT(*) FROM instance_permissions WHERE user_id = ? AND instance_id = ?`
- Error handling: wrap each method in try/catch, log error, return `{ success: false, error: error.message }`
- Return types match the pattern (`auth-database-service.ts`)
