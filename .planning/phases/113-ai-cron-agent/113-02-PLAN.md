<!-- @slide/agent-core AgentRunResult — AgentRunner.run() 的返回类型 -->
interface AgentRunResult {
  finalContent: string | null;
  messages: Message[];
  toolsUsed: string[];
  usage: Record<string, number>;
  stopReason: string;
  error: string | null;
  toolEvents: ToolEvent[];
  hadInjections: boolean;
}

<!-- @slide/agent-core AgentRunSpec — AgentRunner.run() 的参数 -->
interface AgentRunSpec {
  initialMessages: Message[];
  tools: ToolRegistry;
  model: string;
  maxIterations: number;
  maxToolResultChars: number;
  temperature?: number;
  maxTokens?: number;
  reasoningEffort?: string;
  hook: AgentHook;
  failOnToolError?: boolean;
  sessionKey?: string;
  contextWindowTokens?: number;
  llmTimeoutS?: number;
  // ...还有更多可选字段
}

<!-- ToolEvent — AgentHookContext 中的工具调用事件 -->
interface ToolEvent {
  name: string;
  status: "ok" | "error";
  detail: string;
}

<!-- agent-core NoopHook — 空实现的 hook，可继承覆盖 afterIteration -->
class NoopHook implements AgentHook {
  wantsStreaming(): boolean { return false; }
  beforeIteration(ctx: AgentHookContext): Promise<void> | void {}
  onStream(ctx: AgentHookContext, delta: string): Promise<void> | void {}
  onStreamEnd(ctx: AgentHookContext, resuming: boolean): Promise<void> | void {}
  beforeExecuteTools(ctx: AgentHookContext): Promise<void> | void {}
  emitReasoning(text: string | null): Promise<void> | void {}
  emitReasoningEnd(): Promise<void> | void {}
  afterIteration(ctx: AgentHookContext): Promise<void> | void {}
  finalizeContent(ctx: AgentHookContext, content: string | null): string | null { return content; }
}

<!-- complete_analysis.ts 模式 — AnyAgentTool 定义和 toolCatalog.register -->
interface AnyAgentTool {
  name: string;
  description: string;
  parameters: { type: "object"; properties: Record<string, any>; required?: string[] };
  group: string;
  handler: (args: any) => Promise<{ success: boolean; data?: any; error?: string; summary?: string }>;
}

    B) CronExecutor 类：
      - 构造函数接受三个参数（从 @slide/agent-core 导入类型）：
        private runner: AgentRunner (从 @slide/agent-core 导入 AgentRunner)
        private registry: ToolRegistry (从 @slide/agent-core 导入 ToolRegistry, 类型: import('./types').ToolRegistry 或 @slide/agent-core)

      - execute(jobId: number, taskDescription: string, timeoutSeconds: number): Promise<AgentRunResult> 方法：
        1. 生成 sessionKey = `cron:${jobId}:${Date.now()}`
        2. 创建 CronHook 实例
        3. 调用 this.runner.run({ ... }) 传入：
           - initialMessages: [
               { role: 'system', content: buildSystemPrompt(taskDescription) },
               { role: 'user', content: taskDescription }
             ]
           - tools: this.registry
           - model: this.provider.getDefaultModel()
           - maxIterations: 20
           - maxToolResultChars: 20000
           - temperature: 0.0
           - reasoningEffort: 'medium'
           - hook: 上面创建的 CronHook 实例
           - contextWindowTokens: 200_000
           - maxTokens: 4096
           - llmTimeoutS: timeoutSeconds（D-06 超时由 AgentRunner 内部强制）
           - failOnToolError: false
           - sessionKey
        4. 返回 AgentRunResult（从 runner.run 的返回提取，但使用 hook.events 作为 toolEvents）
        5. catch 块：保存 hook.events 作为 partial_trace（从 CronHook.events 导出）
           - 返回 { finalContent: null, error: error.message, toolsUsed: [], toolEvents: hook.events, usage: {}, stopReason: 'error', durationMs: 0 }

      - private buildSystemPrompt(task: string): string
        返回一个数据库运维 Agent system prompt，包含：
        - 角色定义
        - TASK: {task}
        - 约束：5 分钟执行时间、失败重试、自主执行不请求确认
        - 完成时调用 slide_complete_cron
        - 仅允许只读操作，禁止 DDL/DML
        - 工作流建议：检查健康 → 执行诊断 → 分析 → 报告

      - import 模式（从 @slide/agent-core）：
        import { AgentRunner, NoopHook, ToolRegistry } from '@slide/agent-core';
        import type { AgentHook, AgentHookContext, ToolEvent, Message, LLMProvider, AgentRunResult } from '@slide/agent-core';

    D-07 实现说明：CronExecutor 使用 DirectAdapter 的 AgentRunner.run() 直接执行，不经过 Gateway sendGatewayChat()。这是 D-07 在当前项目架构下的正确实现（Phase 108-111 已从 Gateway 迁移到 @slide/agent-core）。

    sessionKey 格式 `cron:${jobId}:${Date.now()}` 满足 D-04 每次执行唯一会话的隔离要求。
    超时通过 llmTimeoutS 和 catch 块中的 hook.events 持久化满足 D-06。
