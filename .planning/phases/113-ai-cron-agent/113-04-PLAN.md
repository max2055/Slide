       private closeDeleteConfirm() {
         this.deleteConfirmOpen = false;

       private async executeDelete() {
         if (!this.deleteConfirmJob) return;
         this.deleteDeleting = true;
         try {
           const res = await authFetch(`/api/cron/jobs/${this.deleteConfirmJob.id}`, { method: "DELETE" });
           if (!res.ok) throw new Error((await res.json().catch(() => ({}))).error || "删除失败");
           this.closeDeleteConfirm();
           this.showToast("已删除");
           this.loadCronJobs();
         } catch (e: any) {

| Boundary | Description |
|----------|-------------|
| User → Form inputs | 用户输入通过 authFetch 发送到后端 API——输入在服务端验证 |
| Frontend → API | POST/PUT/DELETE 请求由服务端 requirePermission 授权 |
| task_description textarea | 用户输入的 NL 文本成为 Agent 的 system prompt——不渲染为 HTML，无 XSS 向量 |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-113-01 | Tampering | task_description textarea | accept | task_description 是管理用户（已认证）填写的 NL 文本，后端验证非空，存到数据库；不渲染为 HTML |
| T-113-02 | Spoofing | Form submission | mitigate | 所有 API 调用都通过 authFetch —— 自动携带 JWT token |
</threat_model>
