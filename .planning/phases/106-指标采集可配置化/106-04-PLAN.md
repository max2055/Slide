    /**
     * AI SQL 生成服务 (D-10)
     * 调用系统配置的 LLM 根据自然语言描述生成采集 SQL
     */
    import { aiAnalysisConfigService } from './ai-analysis-config-service.js';

    /**
     * 生成采集 SQL
     * @param dbType 数据库类型: mysql | postgresql | oracle | dameng
     * @param description 自然语言描述，如"获取当前活跃连接数"
     * @param instanceId 可选的实例 ID，用于 Provider.describeSchema 获取表结构上下文
     * @returns 生成的 SQL 字符串或错误信息
     */
    export async function generateCollectionSql(
      dbType: string,
      description: string,
      instanceId?: number
    ): Promise<{ sql?: string; error?: string }> {
      if (!description || !description.trim()) {
        return { error: '请提供指标描述' };
      }

      try {
        // Build system prompt for SQL generation
        const dbTypeName = { mysql: 'MySQL', postgresql: 'PostgreSQL', oracle: 'Oracle', dameng: '达梦' }[dbType] || dbType;
        const systemPrompt = `你是一个数据库运维专家。请根据用户描述，生成一条 ${dbTypeName} 的监控指标采集 SQL。

    要求：
    1. 仅 SELECT 查询
    2. 返回单行单列的标量值（数字）
    3. 不要包含 DDL/DML
    4. 如果是 deltas/rates，优先使用秒级计算
    5. 查询性能优先，避免全表扫描
    6. 用中文回复，但 SQL 本身用标准语法

    SQL 示例：
    - MySQL: SELECT COUNT(*) as val FROM information_schema.PROCESSLIST
    - PostgreSQL: SELECT count(*) as val FROM pg_stat_activity WHERE state = 'active'
    - Oracle: SELECT COUNT(*) as val FROM V$SESSION WHERE STATUS = 'ACTIVE'
    - Dameng: SELECT COUNT(*) as val FROM V$SESSIONS WHERE STATE = 'ACTIVE'

    只返回 SQL，不要加解释。`;

        const userPrompt = `数据库类型: ${dbTypeName}\n描述: ${description}\n\n生成采集 SQL:`;

        // Call LLM through the existing AI infrastructure
        // Try ai-analysis-config-service.getActiveProvider() first
        const response = await callLLM(systemPrompt, userPrompt);

        // Extract SQL from response (might contain markdown code blocks)
        const sql = extractSqlFromResponse(response);
        if (!sql) {
          return { error: 'AI 未能生成有效的 SQL，请重试或手动编写' };
        }

        return { sql };
      } catch (e: any) {
        return { error: `AI SQL 生成失败: ${e.message}` };
      }
    }

    /**
     * 调用系统 LLM
     * 复用 ai-analysis-config-service 获取配置，调用 LLM API
     */
    async function callLLM(systemPrompt: string, userPrompt: string): Promise<string> {
      // 尝试通过 ai-analysis-config-service 获取 LLM 配置
      // 如果该服务不可用，回退到直接调用 Anthropic/OpenAI
      // 参考 apps/db-ops-api/src/ai-analysis-config-service.ts 和 ai-agent-bridge.ts 的模式
      // 实际实现时可能需要选择 AI provider 类型
      // 简化实现: 调用 LLM 并返回字符串
    }

    function extractSqlFromResponse(response: string): string | null {
      // Strip markdown code blocks if present
      const codeBlockMatch = response.match(/```(?:sql)?\s*([\s\S]*?)```/);
      const sql = codeBlockMatch ? codeBlockMatch[1].trim() : response.trim();
      return sql || null;
    }
    ```

    **[实现细节]** 实际 `callLLM()` 的实现参考现有服务:
    - 首先尝试从 `llmDatabaseService.getDefaultProvider()` 获取默认 LLM provider
    - 然后通过 HTTP 调用 LLM API (Anthropic/OpenAI/Ollama)
    - 或者如果 `aiAgentBridge` 存在，通过 bridge 调用
    - 最简单的方式: 检查 `src/ai-analysis-config-service.ts` 是否有 `getConfig()` 方法获取 LLM endpoint
    - 如果以上都不可用, fallback 到从环境变量读取 `ANTHROPIC_API_KEY` 直接调用 Anthropic SDK

    将 `callLLM` 实现为:
    1. 从 `llmDatabaseService.getDefaultProvider()` 获取 provider
    2. 如果 provider 是 anthropic → 用 `anthropic` SDK 或直接 HTTP POST
    3. 如果 provider 是 openai → 用 `openai` SDK 或直接 HTTP POST
    4. 如果 provider 是 ollama → 直接 HTTP POST 到 ollama endpoint
    5. 响应处理: 从 messages 中提取 content

    简化方案: 如果现有服务太复杂，直接使用 `fetch` 调用 Anthropic API（复用 env `ANTHROPIC_API_KEY`），这是最可靠的路径。但优先使用已配置的 LLM provider。

      }
    });

    private async _aiGenerateSql() {
      const root = this.shadowRoot;
      if (!root) return;
      const statusEl = root.querySelector("#ai-sql-status");
      if (statusEl) statusEl.textContent = '生成中...';

      try {
        // Get the db_type from checkboxes
        const checkedTypes: string[] = [];
        root.querySelectorAll('.checkbox-group input[type="checkbox"]:checked')
          .forEach(cb => checkedTypes.push((cb as HTMLInputElement).value));
        const dbType = checkedTypes[0] || 'mysql';

        // Get the metric description from the name/description field
        const description = (root.querySelector("#metric-desc") as HTMLTextAreaElement).value.trim()
          || (root.querySelector("#metric-name") as HTMLInputElement).value.trim();

        if (!description) {
          if (statusEl) statusEl.textContent = '请先填写指标名称或描述';
          return;
        }

        const res = await fetch(`${API_BASE}/metrics/generate-sql`, {
          method: "POST",
          headers: { "Content-Type": "application/json", Authorization: `Bearer ${this._getToken()}` },
          body: JSON.stringify({ db_type: dbType, description }),
        });

        if (!res.ok) {
          const err = await res.json();
          if (statusEl) statusEl.textContent = err.error || '生成失败';
          return;
        }

        const data = await res.json();
        if (data.sql) {
          const sqlArea = root.querySelector("#metric-collection-sql") as HTMLTextAreaElement;
          if (sqlArea) {
            sqlArea.value = data.sql;
            // Trigger input event for lit to pick up
            sqlArea.dispatchEvent(new Event('input', { bubbles: true }));
          }
          if (statusEl) statusEl.textContent = 'SQL 已生成 ✓';
        } else {
          if (statusEl) statusEl.textContent = data.error || '生成失败';
        }
      } catch (e: any) {
        if (statusEl) statusEl.textContent = `错误: ${e.message}`;
      }
    }
    ```
