      async collect(instance: DatabaseConnection, metricDef: MetricDefinition): Promise<number | null> {
        if (!metricDef.collection_sql) return null;
        const validation = validateSqlIsSelectOnly(metricDef.collection_sql);
        if (!validation.valid) return null;  // validation error logged

        // Enforce single-row single-column scalar
        const sql = `SELECT (${metricDef.collection_sql}) AS val LIMIT 1`;
        try {
          // Execute via instance connection (read-only)
          const result = await this._executeQuery(instance, sql);
          const val = result?.[0]?.val;
          return val != null ? Number(val) : null;
        } catch (e: any) {
          console.error(`[CustomSQLProvider] ${metricDef.id} 采集失败:`, e.message);
          return null;
        }
      }

    class UnifiedCollector {
      /**

       * 从 metric_definitions 读取配置，匹配 Provider，执行采集，记录到 metrics_history
       */
      async collectInstance(instance: any): Promise<void> {
        const dbType = instance.db_type;
        const providers = collectorRegistry.getProvidersByDbType(dbType);
        if (providers.length === 0) return;

        const conn = /* get connection from databaseService */;
        if (!conn) return;

        // Read all metric definitions for this db_type from metric-registry
        const definitions = metricRegistry.getByDbType(dbType)
          .filter(m => m.is_collected);

        const results: Record<string, number> = {};
        let hasData = false;

        for (const def of definitions) {
          const provider = collectorRegistry.get(def.id) || collectors.find(p => p.supportedDbTypes.includes(dbType));
          // First try specific named provider, then fallback to any matching db_type provider
          // Actually: iterating providers by db_type, for each provider check if it can handle this metric
        }

        // Simplified: find a provider that can handle this db_type, then collect all metrics
        for (const provider of providers) {
          if (!provider.enabled) continue;
          for (const def of definitions) {
            try {
              const val = await provider.collect(conn as any, def as any);
              if (val !== null) {
                results[def.id] = val;
              }
            } catch (e: any) {
              const failures = collectorRegistry.recordFailure(provider.name);
              if (failures >= 3) {
                collectorRegistry.disable(provider.name);
                console.error(`[Collector] Provider ${provider.name} 连续失败 ${failures} 次，已自动禁用`);
              }
            }
          }
        }

        // Record to metrics_history
        if (Object.keys(results).length > 0) {
          // Handle built-in metrics (fixed columns) vs dynamic (JSON column)
          const fixedCols: Record<string, number> = {};
          const dynamicData: Record<string, number> = {};
          for (const [key, val] of Object.entries(results)) {
            // Metrics that have fixed columns go to fixedCols
            // Others go to metrics_data JSON
            if (['cpu_usage','memory_usage','disk_usage','connections','qps','tps',

                 'threads_running','threads_connected','bytes_received','bytes_sent',

                 'idx_scan_ratio','dead_tuples','cache_hit_ratio','connections_used','connections_max',
                 'vacuum_count','autovacuum_count','replication_lag_seconds','data_size_gb'].includes(key)) {
              fixedCols[key] = val;
            } else {
              dynamicData[key] = val;
            }
          }
          await metricsDatabaseService.recordMetrics({
            instance_id: instance.id,
            ...fixedCols,
            metrics_data: Object.keys(dynamicData).length > 0 ? dynamicData : undefined,
          } as any);

          // Track success
          for (const m of definitions) {
            collectionCapabilityTracker.recordMetricAttempt(instance.id, m.name, results[m.id] !== undefined);
          }
        }
      }
    }
