**Part C — 新增 DatabaseAuditLogStore 类**

在 `audit-log.ts` 的 `MemoryAuditLogStore` 类之后，新增 `DatabaseAuditLogStore` 类实现 `AuditLogHandler`：

```typescript
class DatabaseAuditLogStore implements AuditLogHandler {
  private pool: mysql.Pool;

  constructor(pool: mysql.Pool) {
    this.pool = pool;
  }

  async write(entry: AuditLogEntry): Promise<void> {
    await this.pool.execute(
      `INSERT INTO sql_execution_history
       (user_id, username, instance_id, instance_name, db_type, database_name,
        sql_text, status, duration_ms, row_count, error_message, ip_address)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [
        entry.userId ?? null,
        entry.username ?? null,
        entry.resourceId ?? null,
        entry.details?.instanceName ?? null,
        entry.details?.dbType ?? null,
        entry.details?.database ?? null,
        entry.details?.sqlText?.substring(0, 5000) ?? null,
        entry.details?.status ?? 'success',
        entry.details?.durationMs ?? 0,
        entry.details?.rowCount ?? 0,
        entry.details?.errorMessage ?? null,
        entry.ip ?? null,
      ]
    );
  }

  async query(filter: AuditLogFilter): Promise<{ entries: AuditLogEntry[]; total: number }> {
    const conditions: string[] = [];
    const params: any[] = [];

    if (filter.eventType) {
      conditions.push('e.event_type = ?');
      params.push(filter.eventType);
    }
    if (filter.userId) {
      conditions.push('e.user_id = ?');
      params.push(filter.userId);
    }
    if (filter.resourceId) {
      conditions.push('e.instance_id = ?');
      params.push(filter.resourceId);
    }
    if (filter.startTime) {
      conditions.push('e.created_at >= ?');
      params.push(filter.startTime);
    }
    if (filter.endTime) {
      conditions.push('e.created_at <= ?');
      params.push(filter.endTime);
    }
    if (filter.search) {
      conditions.push('e.sql_text LIKE ?');
      params.push(`%${filter.search}%`);
    }

    const where = conditions.length > 0 ? `WHERE ${conditions.join(' AND ')}` : '';

    const [countRows] = await this.pool.query(
      `SELECT COUNT(*) as total FROM sql_execution_history e ${where}`,
      params
    ) as any;
    const total = countRows[0]?.total ?? 0;

    const limit = filter.limit ?? 50;
    const offset = filter.offset ?? 0;

    const [rows] = await this.pool.query(
      `SELECT e.* FROM sql_execution_history e ${where} ORDER BY e.id DESC LIMIT ? OFFSET ?`,
      [...params, limit, offset]
    ) as any;

    const entries: AuditLogEntry[] = (rows as any[]).map((row: any) => ({
      id: row.id,
      timestamp: row.created_at instanceof Date ? row.created_at.toISOString() : String(row.created_at),
      eventType: filter.eventType || 'sql_execution',
      userId: row.user_id,
      username: row.username || undefined,
      resourceId: row.instance_id,
      ip: row.ip_address,
      details: {
        instanceName: row.instance_name,
        dbType: row.db_type,
        database: row.database_name,
        sqlText: row.sql_text,
        status: row.status,
        durationMs: row.duration_ms,
        rowCount: row.row_count,
        errorMessage: row.error_message,
      },
    }));

    return { entries, total };
  }

  async export(filter: AuditLogFilter): Promise<AuditLogEntry[]> {
    const result = await this.query({ ...filter, limit: 10000, offset: 0 });
    return result.entries;
  }
}
```

注意：`AuditLogEntry` 字段映射需要符合现有 interface 定义。检查 `audit-log.ts` 中 `AuditLogEntry` 的字段名——上面使用了 `details.instanceName`、`details.dbType` 等，需要与 `AuditLogManager.logSqlExecution()` 中的 `details` 对象字段名保持一致。

**Part A — AuditLogManager 支持注入 DatabaseAuditLogStore**

在 `AuditLogManager` 中增加一个使用 `DatabaseAuditLogStore` 的方式。两种方案：

方案 1（推荐）：在 `AuditLogManager` 构造函数中同时接受两个 store（memory + database），write 时双写，query/export 时读 database。好处：不影响现有行为。
方案 2：完全替换为 DatabaseAuditLogStore。

采用方案 1 — 双写到内存和数据库，查询从数据库读：

```typescript
class AuditLogManager {
  private primaryStore: AuditLogHandler;      // MemoryAuditLogStore
  private persistentStore?: AuditLogHandler;  // DatabaseAuditLogStore
  ...

  constructor(primaryStore: AuditLogHandler, persistentStore?: AuditLogHandler) {
    this.primaryStore = primaryStore;
    this.persistentStore = persistentStore;
  }

  async logSqlExecution(...): Promise<void> {
    // 现有逻辑：写入 primaryStore（内存）
    const entry: AuditLogEntry = {
      id: ++this.eventCounter,
      timestamp: new Date().toISOString(),
      eventType: 'sql_execution',
      ...
    };
    await this.primaryStore.write(entry);
    // 新增：同时写入持久化 store
    if (this.persistentStore) {
      await this.persistentStore.write(entry).catch(err => {
        console.error('[AuditLogManager] 持久化 SQL 历史失败:', err);
      });
    }
  }

  async queryAuditLogs(filter: AuditLogFilter): Promise<{ entries: AuditLogEntry[]; total: number }> {
    if (this.persistentStore) {
      return this.persistentStore.query(filter);
    }
    return this.primaryStore.query(filter);
  }
}
```

    total: result.total,
