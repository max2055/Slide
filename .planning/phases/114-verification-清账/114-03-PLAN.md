    async findActiveAlert(instanceId: number, metricName: string, ruleId: number): Promise<Alert | null> {
      const [rows] = await pool.query<AlertRow[]>(
        `SELECT * FROM alerts 
         WHERE instance_id = ? AND metric_name = ? AND rule_id = ? 
         AND status IN ('unread', 'read', 'acknowledged')
         ORDER BY created_at DESC LIMIT 1`,
        [instanceId, metricName, ruleId]
      );
      return rows.length > 0 ? this._rowToAlert(rows[0]) : null;
    }

    async touchAlert(alertId: number, metricValue: number): Promise<void> {
      await pool.query(
        `UPDATE alerts SET metric_value = ?, updated_at = NOW() WHERE id = ?`,
        [metricValue, alertId]
      );
    }
    ```

    async getActiveAlerts(): Promise<Alert[]> {
      const [rows] = await pool.query<AlertRow[]>(

    }

    export function isValueHealthy(value: number, rule: AlertRule): boolean {
      const tpl = rule.threshold_template;
      if (!tpl) return true; // no threshold = always healthy
      
      const checkValue = (level: string): boolean => {
        const threshold = tpl[level as keyof typeof tpl];
        if (threshold == null) return true;
        const parsed = parseThreshold(threshold);
        if (parsed === null) return true;
        
        const op = rule.operator ?? '>=';
        switch (op) {
          case '>=': return value >= parsed;
          case '>': return value > parsed;
          case '<=': return value <= parsed;
          case '<': return value < parsed;
          case '==': return value === parsed;
          default: return true;
        }
      };
      
      // Healthy = does NOT trigger any level
      return !checkValue('critical') && !checkValue('error') && !checkValue('warning');
    }
    ```
    (Adjust based on actual threshold_template structure and operator field.)

       const activeAlerts = await alertDatabaseService.getActiveAlerts();

         try {

           if (!metrics) continue; // instance unreachable, skip
