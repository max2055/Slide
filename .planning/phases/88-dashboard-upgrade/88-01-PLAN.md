    Route 1 — `/api/dashboard/capacity-trend` (immediately after existing capacity history route):

    This endpoint accepts optional query parameters:
    - `hours` (number, default 168): lookback window in hours
    - `instance_id` (number, optional): filter to a single instance
    - `start_date` (string, optional, YYYY-MM-DD): custom start date (overrides hours when provided with end_date)

    ```typescript
    // Dashboard: aggregated data volume trend (DASH-02 / D-07)
    // Supports instance_id (D-06) and start_date/end_date (D-05) filtering
    fastify.get('/api/dashboard/capacity-trend', async (request, reply) => {
      try {
        const query = request.query as any;
        const hours = Number(query?.hours) || 168;
        const instance_id = query?.instance_id ? Number(query.instance_id) : null;
        const start_date = query?.start_date || null;
        const end_date = query?.end_date || null;
        const pool = dbConnection.getPool();
        if (!pool) return reply.code(500).send({ error: '数据库未连接' });

        // Build WHERE clause: support hours-based OR date-range-based filtering
        let whereClause = '';
        const params: any[] = [];

        if (start_date && end_date) {
          whereClause = 'WHERE recorded_at >= ? AND recorded_at < DATE_ADD(?, INTERVAL 1 DAY)';
          params.push(start_date, end_date);
        } else {
          whereClause = 'WHERE recorded_at >= NOW() - INTERVAL ? HOUR';
          params.push(hours);
        }

        if (instance_id) {
          whereClause += ' AND instance_id = ?';
          params.push(instance_id);
        }

        // Cross-instance aggregation with hour-level bucket
        const [rows] = await pool.execute(
          `SELECT
             DATE_FORMAT(recorded_at, '%Y-%m-%d %H:00:00') as time_bucket,
             SUM(total_size_gb) as total_size_gb,
             COUNT(DISTINCT instance_id) as instance_count
           FROM capacity_history
           ${whereClause}
           GROUP BY DATE_FORMAT(recorded_at, '%Y-%m-%d %H:00:00')
           ORDER BY time_bucket ASC`,
          params
        ) as any;

        // Current total — per-instance or aggregated across all
        let currentTotal = 0;
        if (instance_id) {
          const [current] = await pool.execute(
            `SELECT total_size_gb as current_total
             FROM capacity_history
             WHERE instance_id = ?
             ORDER BY recorded_at DESC
             LIMIT 1`,
            [instance_id]
          ) as any;
          currentTotal = Number(current[0]?.current_total || 0);
        } else {
          const [current] = await pool.execute(
            `SELECT SUM(ch.total_size_gb) as current_total
             FROM capacity_history ch
             INNER JOIN (
               SELECT instance_id, MAX(recorded_at) as max_time
               FROM capacity_history
               GROUP BY instance_id
             ) latest ON ch.instance_id = latest.instance_id AND ch.recorded_at = latest.max_time`,
          ) as any;
          currentTotal = Number(current[0]?.current_total || 0);
        }

        reply.send({
          current_total_gb: currentTotal,
          trend: rows.map((r: any) => ({
            time: r.time_bucket,
            total_size_gb: Number(r.total_size_gb),
            instance_count: instance_id ? 1 : r.instance_count,
          })),
        });
      } catch (error: any) {
        reply.code(500).send({ error: error.message });
      }
    });
    ```

    Route 2 — `/api/dashboard/ai-stats` (after capacity-trend):

    ```typescript
    // Dashboard: AI analysis daily count (DASH-03 / D-15)
    fastify.get('/api/dashboard/ai-stats', async (request, reply) => {
      try {
        const pool = dbConnection.getPool();
        if (!pool) return reply.code(500).send({ error: '数据库未连接' });

        const [rows] = await pool.execute(
          `SELECT COUNT(*) as cnt, analysis_type
           FROM ai_analysis
           WHERE created_at >= CURDATE()
           GROUP BY analysis_type
           UNION ALL
           SELECT COUNT(*) as cnt, NULL as analysis_type
           FROM ai_analysis
           WHERE created_at >= CURDATE()`,
        ) as any;

        let today_total = 0;
        const breakdown: Record<string, number> = {};

        for (const r of rows) {
          if (r.analysis_type === null) {
            today_total = Number(r.cnt);
          } else {
            breakdown[r.analysis_type] = Number(r.cnt);
          }
        }

        reply.send({
          today_total,
          breakdown,
          last_updated: new Date().toISOString(),
        });
      } catch (error: any) {
        reply.code(500).send({ error: error.message });
      }
    });
    ```

| T-88-03 | Information Disclosure | `/api/dashboard/capacity-trend` | accept | The endpoint returns aggregated totals across all instances. While an attacker could learn total data volume, this is the same information available from individual instance capacity endpoints. No PII or sensitive metadata is exposed. |
