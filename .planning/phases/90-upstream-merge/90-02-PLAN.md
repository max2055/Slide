<task type="auto">
  <name>Task 1: Start gateway and backend, verify health endpoints</name>
  <files></files>
  <action>
    1. Kill any existing processes on gateway and backend ports:
       ```
       lsof -ti:28789 | xargs kill -9 2>/dev/null || true
       lsof -ti:3000 | xargs kill -9 2>/dev/null || true
       ```
    2. Start Gateway:
       - Run `npx tsx openclaw.mjs 2>&1 &`
       - Or if gateway requires build: `node scripts/run-node.mjs &`
       - Wait 15 seconds for startup
       - Check gateway is running: `lsof -ti:28789 | head -1`
    3. Start Backend (db-ops-api):
       - `cd apps/db-ops-api && npx tsx server.ts &`
       - Wait 10 seconds for startup
       - Check backend health: `curl -s -o /dev/null -w "%{http_code}" http://localhost:3000/api/health`
    4. Verify both services are reachable:
       - Gateway PID: `lsof -ti:28789`
       - Backend health endpoint returns 200
  </action>
  <verify>
    <automated>
      echo "=== Gateway ==="
      GW_PID=$(lsof -ti:28789 2>/dev/null | head -1)
      echo "Gateway PID: ${GW_PID:-NOT RUNNING}"
      test -n "$GW_PID" || exit 1
      echo "=== Backend Health ==="
      HEALTH=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:3000/api/health 2>/dev/null || echo "FAIL")
      echo "Health: $HEALTH"
      test "$HEALTH" = "200" || exit 1
      echo "BOTH SERVICES ARE RUNNING"
    </automated>
  </verify>
  <done>
    - Gateway is running on port 28789
    - Backend health endpoint returns HTTP 200
  </done>
