    ```typescript
    async function executeSqlOptimization(params: Record<string, unknown>): Promise<ToolCallResult> {
      try {
        const sql = params.sql as string | undefined;
        const instanceId = params.instance_id ? Number(params.instance_id) : undefined;
        const limit = params.limit ? Math.min(Number(params.limit), 50) : 10;

        // Mode 2: 查询慢查询列表（不传 sql 时）
        if (!sql) {
          if (!instanceId) {
            // 没有指定实例 — 获取所有实例列表供 LLM 选择
            const instances = await instanceDatabaseService.getAllInstances();
            if (!instances || instances.length === 0) {
              return { success: true, data: { message: '数据库中没有配置实例，无法查询慢查询' } };
            }
            if (instances.length === 1) {
              // 只有一个实例，自动查询
              const slowQueries = await metricsDatabaseService.getSlowQueries(instances[0].id, limit);
              return {
                success: true,
                data: {
                  instance_id: instances[0].id,
                  instance_name: instances[0].name,
                  count: slowQueries.length,
                  slow_queries: slowQueries.map(sq => ({
                    id: sq.id,
                    sql_text: sq.sql_text ? sq.sql_text.substring(0, 200) : '',
                    avg_time_ms: sq.avg_time_ms,
                    max_time_ms: sq.max_time_ms,
                    execution_count: sq.execution_count,
                    rows_examined: sq.rows_examined,
                    rows_sent: sq.rows_sent,
                    first_seen: sq.first_seen,
                    last_seen: sq.last_seen,
                    schema_name: sq.schema_name,
                    digest_text: sq.digest_text,
                  })),
                },
              };
            }
            // 多个实例时，让 LLM 选择
            return {
              success: true,
              data: {
                message: '请指定要查询的实例 ID',
                available_instances: instances.map(i => ({ id: i.id, name: i.name, db_type: i.db_type })),
              },
            };
          }

          // 有 instance_id，查询指定实例的慢查询
          const slowQueries = await metricsDatabaseService.getSlowQueries(instanceId, limit);
          return {
            success: true,
            data: {
              instance_id: instanceId,
              count: slowQueries.length,
              slow_queries: slowQueries.map(sq => ({
                id: sq.id,
                sql_text: sq.sql_text ? sq.sql_text.substring(0, 200) : '',
                avg_time_ms: sq.avg_time_ms,
                max_time_ms: sq.max_time_ms,
                execution_count: sq.execution_count,
                rows_examined: sq.rows_examined,
                rows_sent: sq.rows_sent,
                first_seen: sq.first_seen,
                last_seen: sq.last_seen,
                schema_name: sq.schema_name,
                digest_text: sq.digest_text,
              })),
            },
          };
        }

        // Mode 1: 分析指定 SQL（当前返回基础格式，未来可增强）
        return {
          success: true,
          data: {
            sql,
            instance_id: instanceId,
            analysis: `正在分析 SQL 语句…\n\nSQL: ${sql.substring(0, 500)}`,
          },
        };
      } catch (error: any) {
        return { success: false, error: `慢查询分析失败: ${error.message}` };
      }
    }
    ```
