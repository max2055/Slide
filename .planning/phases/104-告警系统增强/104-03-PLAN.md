private async _toggleConfig(cfg: ReportConfig) {
  const originalEnabled = cfg.enabled;
  cfg.enabled = !cfg.enabled;      // optimistic update
  this.requestUpdate();
  try {
    const res = await authFetch(`/api/reports/configs/${cfg.id}`, {
      method: 'PUT',
      headers: this.headers,
      body: JSON.stringify({ enabled: cfg.enabled }),
    });
    if (!res.ok) throw new Error('update failed');
    this.loadConfigs();
  } catch (err) {
    cfg.enabled = originalEnabled; // rollback
    this.requestUpdate();
  }
}

    private _validateThresholds(t: { warning?: number | null; error?: number | null; critical?: number | null }): boolean {
      const w = t.warning;
      const e = t.error;
      const c = t.critical;
      // All empty = valid (user hasn't set any yet)
      if (w == null && e == null && c == null) {
        this._thresholdValidationError = null;
        return true;
      }
      // Check ordering: warning < error < critical (only check non-null adjacent pairs)
      if (w != null && e != null && w >= e) {
        this._thresholdValidationError = 'Warning 必须小于 Error';
        return false;
      }
      if (e != null && c != null && e >= c) {
        this._thresholdValidationError = 'Error 必须小于 Critical';
        return false;
      }
      if (w != null && c != null && w >= c) {
        this._thresholdValidationError = 'Warning 必须小于 Critical';
        return false;
      }
      this._thresholdValidationError = null;
      return true;
    }
    ```

    private async _toggleRuleEnabled(rule: AlertRule) {
      const originalEnabled = rule.enabled;
      rule.enabled = !rule.enabled;  // optimistic update
      this.requestUpdate();

      try {
        const res = await authFetch(`/api/alert-rules/${rule.id}`, {
          method: 'PUT',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ enabled: rule.enabled }),
        });
        if (!res.ok) throw new Error('toggle failed');
      } catch (err) {
        rule.enabled = originalEnabled;  // rollback
        this.requestUpdate();
        // Show inline error feedback
        const feedbackEl = this.renderRoot?.querySelector(`[data-feedback-rule="${rule.id}"]`);
        if (feedbackEl) {
          feedbackEl.textContent = '切换失败';
          (feedbackEl as HTMLElement).style.display = 'block';
          setTimeout(() => {
            (feedbackEl as HTMLElement).style.display = 'none';
          }, 3000);
        }
      }
    }
    ```
