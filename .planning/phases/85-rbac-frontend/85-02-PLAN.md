    ```typescript
    private async _openInstancePermModal(user: UserInfo) {
      this.instancePermUser = user;
      this.instancePermLoading = true;
      this.instancePermError = null;
      this.showInstancePermModal = true;
      this.instanceSearchQuery = "";
      try {
        const token = this._getToken();
        const [allInstances, grantedIds] = await Promise.all([
          fetch("/api/database/instances").then(r => r.json()),
          fetch(`/api/v1/rbac/users/${user.id}/instances`, {
            headers: { "Authorization": `Bearer ${token}` },
          }).then(r => {
            if (!r.ok) throw new Error(`加载失败 (${r.status})`);
            return r.json();
          }),
        ]);
        this.allInstances = Array.isArray(allInstances) ? allInstances : (allInstances.instances || []);
        // grantedIds is typically an array of IDs or objects with id field
        const ids = Array.isArray(grantedIds)
          ? grantedIds.map((i: any) => typeof i === 'number' ? i : i.id)
          : [];
        this.grantedInstanceIds = new Set(ids);
        this.savedGrantedInstanceIds = new Set(ids);
      } catch (e: any) {
        this.instancePermError = e.message;
      } finally {
        this.instancePermLoading = false;
      }
    }
    ```

    Add save method:
    ```typescript
    private async _saveInstancePerms() {
      if (!this.instancePermUser) return;
      this.instancePermSaving = true;
      this.instancePermError = null;
      try {
        const token = this._getToken();
        const toGrant = [...this.grantedInstanceIds].filter(
          id => !this.savedGrantedInstanceIds.has(id)
        );
        const toRevoke = [...this.savedGrantedInstanceIds].filter(
          id => !this.grantedInstanceIds.has(id)
        );
        for (const instanceId of toGrant) {
          const res = await fetch(`/api/v1/rbac/users/${this.instancePermUser.id}/instances`, {
            method: "POST",
            headers: { "Authorization": `Bearer ${token}`, "Content-Type": "application/json" },
            body: JSON.stringify({ instanceId }),
          });
          if (!res.ok) throw new Error(`授权失败 (${res.status})`);
        }
        for (const instanceId of toRevoke) {
          const res = await fetch(`/api/v1/rbac/users/${this.instancePermUser.id}/instances/${instanceId}`, {
            method: "DELETE",
            headers: { "Authorization": `Bearer ${token}` },
          });
          if (!res.ok) throw new Error(`撤销失败 (${res.status})`);
        }
        this.savedGrantedInstanceIds = new Set(this.grantedInstanceIds);
        this.showInstancePermModal = false;
      } catch (e: any) {
        this.instancePermError = e.message;
      } finally {
        this.instancePermSaving = false;
      }
    }
    ```

      return html`
        <div class="modal-overlay" @click=${(e: any) => {

        }}>

            </div>

              ${this.instancePermError ? html`<div class="error-msg">${this.instancePermError}</div>` : ""}
